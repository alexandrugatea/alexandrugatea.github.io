8. REDUCE: Sum up the instances of each of these

    
      const data = ['car', 'car', 'truck', 'truck', 'bike',
      'walk', 'car', 'van', 'bike', 'walk', 'car', 'van', 'car', 'truck'];
      // Input
      const instances = data.reduce((object, item) => {
        if (!object[item]) {
          object[item] = 0;
        }
        object[item]++;
        return object;
      }, {});
